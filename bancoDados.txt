CREATE DATABASE planilhas;

USE planilhas;

CREATE TABLE escola(
	id_escola INT NOT NULL AUTO_INCREMENT,
	nome_escola VARCHAR(150),
    codigo VARCHAR(11),
    ead BOOLEAN,
    site VARCHAR(120),
    email VARCHAR(120),
    PRIMARY KEY(id_escola)
);


CREATE TABLE localizacao(
    id_local INT NOT NULL AUTO_INCREMENT,
    escola_id INT NOT NULL,
    rua VARCHAR(150),
    estado CHAR(2),
    cidade VARCHAR(150),
    cep CHAR(8),
    bairro VARCHAR(120),
    PRIMARY KEY(id_local),
    CONSTRAINT local_fk_escola FOREIGN KEY (escola_id) REFERENCES escola(id_escola)
);

CREATE TABLE contatos(
	id_ctt INT NOT NULL AUTO_INCREMENT,
    escola_id INT NOT NULL,
	fone VARCHAR(15),
    PRIMARY KEY(id_ctt),
    CONSTRAINT fone_fk_escola FOREIGN KEY (escola_id) REFERENCES escola(id_escola)
);

CREATE TABLE cargos(
	id_cargos INT NOT NULL AUTO_INCREMENT,
	cargo VARCHAR(150),
    PRIMARY KEY(id_cargos)
);

CREATE TABLE funcionarios(
	id_func INT NOT NULL AUTO_INCREMENT,
    escola_id INT NOT NULL,
    cargos_id INT NOT NULL,
	nome_funcionario VARCHAR(150),
    PRIMARY KEY(id_func)
);
